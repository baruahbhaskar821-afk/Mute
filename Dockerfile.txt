FROM php:8.2-cli

Install system dependencies

RUN apt-get update && apt-get install -y
git
curl
libzip-dev
zip
unzip
supervisor
&& rm -rf /var/lib/apt/lists/*

Install PHP extensions

RUN docker-php-ext-install zip

Install Composer

COPY –from=composer:latest /usr/bin/composer /usr/bin/composer

Set working directory

WORKDIR /var/www/html

Copy application files

COPY . .

Install Python and required packages

RUN apt-get update && apt-get install -y
python3
python3-pip
&& rm -rf /var/lib/apt/lists/*

Install Python dependencies

RUN pip3 install python-telegram-bot==20.7

Create directories for logs and data

RUN mkdir -p /var/log/supervisor /var/www/html/data RUN touch
/var/www/html/geto.log

Set permissions

RUN chmod 755 /var/www/html RUN chmod 644 /var/www/html/.php 2>/dev/null
|| true RUN chmod 644 /var/www/html/.json 2>/dev/null || true RUN chmod
644 /var/www/html/.log 2>/dev/null || true RUN chmod 755
/var/www/html/data RUN chmod 666 /var/www/html/data/.json 2>/dev/null ||
true

Copy supervisor configuration

COPY supervisord.conf /etc/supervisor/conf.d/supervisord.conf

Expose port (for health check, not actually used)

EXPOSE 10000

Start supervisor

CMD [“/usr/bin/supervisord”, “-n”, “-c”,
“/etc/supervisor/conf.d/supervisord.conf”]
